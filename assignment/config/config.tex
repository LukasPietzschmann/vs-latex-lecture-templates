					\tikzset{every on foreground layer/.try,#1}%
					\tikz@options%
				},
			execute at end scope={\endpgfonlayer}
		},

			\pgfonlayer{#1}\begingroup

			\gdef\node@@on@layer{%
				\setbox\tikz@tempbox=\hbox\bgroup\pgfonlayer{#1}\unhbox\tikz@tempbox\endpgfonlayer\egroup}

		},
}
